function fmtDur(m){
  if(!m)return'';
  if(m<60)return`${m}min`;
  const h=m/60;
  return`${Number.isInteger(h)?h:h.toFixed(1)}h`;
}
function allActs(city,tod){
  return[...(ACTS[city]?.[tod]||[]),...(G.custom[city]||[]).filter(a=>a.tod===tod)];
}

// STORAGE  (window.storage → google.script.run → in-memory)
// ══════════════════════════════════════════════════════════════════════
const HAS_WS =typeof window!=='undefined'&&typeof window.storage!=='undefined';
const HAS_GAS=typeof google!=='undefined'&&!!google.script;

async function loadStorage(){
  if(HAS_WS){
    for(const[k,dest]of[['jpkr_slots','slots'],['jpkr_custom','custom'],['jpkr_secs','secs']]){
      try{
        const r=await window.storage.get(k);
        if(r&&r.value){
          const d=JSON.parse(r.value);
          if(dest==='secs'){for(const[sk,v]of Object.entries(d)){if(G.secs[sk])Object.assign(G.secs[sk],v);}}
          else{G[dest]=d||{};}
        }
      }catch{}
    }
  }else if(HAS_GAS){
    await new Promise(res=>{
      google.script.run
        .withSuccessHandler(j=>{
          try{
            const d=JSON.parse(j);
            if(d.slots)G.slots=d.slots;
            if(d.custom)G.custom=d.custom;
            if(d.secs)for(const[k,v]of Object.entries(d.secs)){if(G.secs[k])Object.assign(G.secs[k],v);}
          }catch{}res();
        })
        .withFailureHandler(()=>res())
        .getState();
    });
  }
  // Sanitise — handle migration from old format (had 'route' instead of 'rows')
  for(const sec of Object.keys(CFG)){
    const s=G.secs[sec];
    if(!s.rows||!Array.isArray(s.rows)||!s.rows.length){
      s.rows=mkDefRows(sec);s.ctr=CFG[sec].defRows.length;
    }
    if(!s.ctr)s.ctr=s.rows.length;
    delete s.route;
  }
}

let _st=null;
function scheduleSave(){clearTimeout(_st);_st=setTimeout(doSave,900);}
async function doSave(){
  if(HAS_WS){
    try{
      await window.storage.set('jpkr_slots', JSON.stringify(G.slots));
      await window.storage.set('jpkr_custom',JSON.stringify(G.custom));
      await window.storage.set('jpkr_secs',  JSON.stringify(G.secs));
    }catch{}
  }else if(HAS_GAS){
    google.script.run.withFailureHandler(console.error)
      .saveState(JSON.stringify(G.slots),JSON.stringify(G.custom),JSON.stringify(G.secs));
  }
  const t=document.getElementById('toast');
  t.classList.add('on');setTimeout(()=>t.classList.remove('on'),1800);
}

function openPicker(sec,rowId,tod){
  const row=G.secs[sec].rows.find(r=>r.id===rowId);
  if(!row)return;
  const city=row.c,[bg,,dark]=CLR[city];

  document.getElementById('picker-sub').textContent=todL[tod];
  document.getElementById('picker-hd').style.background=bg;
  _pActs=allActs(city,tod);
  _pState={sec,rowId,tod};
  const cur=G.slots[skey(sec,rowId,tod)];
  const body=document.getElementById('picker-body');
  body.innerHTML=
    (_pActs.length===0?'<p style="color:#9A9590;font-size:13px;padding:16px 0">No activities yet — add one below!</p>':'')+

      <span class="pn">${esc(a.name)}</span>
      <span class="pdur">${fmtDur(a.dur)}</span>
    </div>`).join('')+
    `<div style="margin-top:12px;padding-top:10px;border-top:1px solid #F0EDE8">
      <button class="addb" style="width:100%;margin-top:0" onclick="openAddFromPicker('${city}','${tod}')">+ Add custom activity</button>

}

  document.getElementById('city-modal-sub').textContent=`Day ${vi+1} · ${dateLbl(G.secs[sec].start,vi)}`;
  _cpState={sec,rowId};
  document.getElementById('city-modal-body').innerHTML=CFG[sec].cities.map(city=>{
    const[bg,,dark]=CLR[city];
    const isSel=city===row.c;
    return`<div class="city-opt${isSel?' sel':''}" style="background:${bg};${isSel?`border-color:${dark}`:'border-color:transparent'}"
               onclick="changeCityForRow('${sec}','${rowId}','${city}')">
      <span class="co-name" style="color:${dark}">${LBL[city]}</span>
      ${isSel?`<span style="margin-left:auto;font-size:12px;color:${dark}">✓</span>`:''}
    </div>`;
  }).join('');
  document.getElementById('city-modal').classList.remove('hd');
}

function closeCityPicker(){document.getElementById('city-modal').classList.add('hd');_cpState=null;}

function changeCityForRow(sec,rowId,newCity){
  const row=G.secs[sec].rows.find(r=>r.id===rowId);
  if(!row){closeCityPicker();return;}
  if(row.c!==newCity){
    for(const tod of['morning','midday','evening'])delete G.slots[skey(sec,rowId,tod)];
    row.c=newCity;
    renderRows(sec);
    renderBank(sec);
    scheduleSave();
  }
  closeCityPicker();
}

// ROW MANAGEMENT
// ══════════════════════════════════════════════════════════════════════
function addRow(sec){
  const s=G.secs[sec];
  const lastCity=s.rows.length?s.rows[s.rows.length-1].c:CFG[sec].cities[0];
  s.rows.push({id:`r${s.ctr++}`,c:lastCity});
  renderRows(sec);scheduleSave();
}

function removeRow(e,sec,rowId){
  e.stopPropagation();
  const s=G.secs[sec];
  if(s.rows.length<=1)return;
  for(const tod of['morning','midday','evening'])delete G.slots[skey(sec,rowId,tod)];
  s.rows=s.rows.filter(r=>r.id!==rowId);
  renderRows(sec);scheduleSave();
}

function submitCustomAct(){
  const city=document.getElementById('add-city-sel').value;
  const tod =document.getElementById('add-tod-sel').value;
  const emoji=(document.getElementById('add-emoji-inp').value.trim()||'📍');
  const name =document.getElementById('add-name-inp').value.trim();
  const dur  =parseInt(document.getElementById('add-dur-inp').value)||0;
  const desc =document.getElementById('add-desc-inp').value.trim();

  G.custom[city].push({emoji,name,tod,dur,desc});
  closeAddModal();

  scheduleSave();
}
